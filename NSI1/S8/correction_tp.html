def dataset_petale():
    """ 
    Cree a partir de iris une table (tableau de tableaux) de 150 lignes avec 3 colonnes.
    Chaque ligne represente (longueur du petale, largeur du petale, nom de l'espece) pour une fleur du jeu de donnees.

    On sait que: 
    - iris['data'] est une table de 150 lignes avec 4 colonnes: 
    (sepal length, sepal width, petal length, petal width)
    - iris['target'] est un tableau de 150 valeurs : 
    (code de la fleur), qui vaut 0 si la fleur est 'setosa', 1 si 'versicolor', 2 si 'virginica'.

    Donc pour chaque i dans 0,1,...,149:
    - iris['data'][i][2] donne la longueur du petale de la fleur i.
    - especedict[iris['target'][i]] est l'espece de la fleur 


    REMARQUE: les programmes d'apprentissages vont assez souvent distinguer 
    - le tableau des données (on le note souvent X dans sklearn)
    - le tableau des valeurs que l'on veut predire, donc ici l'espece (on le note souvent y)
    Nous avons choisi d'inclure les donnees et valeurs a predire dans un meme tableau,
    mais ce choix n'est pas forcement le plus courant. 
    """
    # On utilise le tableau ci-dessous comme un dictionnaire, 
    # pour remplacer le code de l'espece (0,1,2) par le nom correspondant.
    especedict = ['setosa', 'versicolor', 'virginica']
    return [ 
        [iris['data'][i][2], 
        iris['data'][i][3], 
        especedict[iris['target'][i]]] 
        for i in range(150) ]


def illustre_petales(dataset_2d_espece):
    """ 
    Entree: un tableau avec 3 colonnes
    Chaque ligne represente (longueur du petale, largeur du petale, nom de l'espece) pour une fleur du jeu de donnees.

    Affiche dans figure un point par fleur:
    - sur l'axe des x la characteristique 0 (longueur petale)
    - sur l'axe des y la characteristique 1 (largeur petale)
    - la couleur indique l'espece de la fleur

    """
    # On utilise les 2 tableaux ci-dessous comme s'ils etaient des dictionnaires, 
    # pour associer a un entier (0,1,2) une couleur ou espece.
    couleurdict = {'setosa':'r', 'versicolor':'g', 'virginica':'b', 'inconnu':'k'}
    markerdict = {'setosa':'1', 'versicolor':'2', 'virginica':'3', 'inconnu':'o'}
    for espece in ['setosa', 'versicolor', 'virginica', 'inconnu']:
        # on trace separement les points pour chaque espece.
        # avec matplotlib3 on pourrait directement tracer les 3 
        # en utilisant une seule instruction scatter, avec un tableau de couleurs 
        # et en utilisant legend_elements pour associer les labels
        x = [ t[0] for t in dataset_2d_espece if t[2] == espece ]
        y = [ t[1] for t in dataset_2d_espece if t[2] == espece ]
        couleur = couleurdict[espece]
        marqueur = markerdict[espece]
        plt.scatter(x, y, c=couleur, marker = marqueur, label=espece)
    plt.xlabel("longueur petale")
    plt.ylabel("largeur petale")
    plt.legend()

    from math import sqrt

def distance_euclidienne(point1, point2, n):
    """ 
    Suppose que point1 et point2 sont deux tableaux de longueur au moins n.
    Renvoie la distance euclidienne entre point1 et point2 en prenant seulement en compte leurs coordonnees <= n.
    /!\ Ajout: n est le nombre de coordonnées à prendre en compte pour le calcul de distance car ...
    (on ne veut en general pas prendre la derniere coordonnee qui peut etre une etiquette).
    Complexite: lineaire.

    On rappelle que si 
    point1 = [x1, ..., xn]
    point2 = [y1, ..., yn]
    la distance euclidienne est sqrt((y1-x1)**2 + (y2-x2)**2 + ... (yn-xn)**2)
    """
    assert(len(point1) >= n and len(point2) >= n)
    tot = 0
    for i in range(n):
        tot += (point1[i]-point2[i])**2
    return sqrt(tot)

    def plus_proches_voisins(training, p, k):
    """ 
    Entree: 
    - training: tableau de données d'entrainement contenant m points,
    - p: inconnue, tableau de longueur d, 

    - k: entier naturel strictement positif. 

    Renvoie le tableau des k plus proches voisins de p parmi training
    pour la distance euclidienne sur les d-1 premieres coordonnes.

    Complexite: O(m * log(m) * d) pour cette version simplifiee.
    Si on calculait toutes les distances avant de trier, 
    la complexite deviendrait O(m * log(m) + m * d).
    """
    assert(0 < k and k < len(training))

    def distance(pt):
        """renvoie la distance entre p et pt"""
        assert(len(pt) >= len(p) - 1)
        return distance_euclidienne(pt, p, len(p) - 1)

    training_trie = sorted(training, key=distance) # le critère de tri est fourni par la fonction distance
    return [training_trie[i] for i in range(k)]

def plus_proches_voisins2(training, p, k):
    """ 
    Entree: 
    - training: tableau de données d'entrainement contenant m points,
    - p: inconnue, tableau de longueur d, 

    - k: entier naturel strictement positif. 

    Renvoie le tableau des k plus proches voisins de p parmi training
    pour la distance euclidienne sur les d-1 premieres coordonnes.

    Complexite: O(m * k * d) pour cette version simplifiee.
    Si on calculait toutes les distances avant de trier, 
    la complexite deviendrait O(m * k + m * d).
    """
    assert(0 < k and k < len(training))
    t = training.copy() # une copie superficielle suffit
    for i in range(k):
        # On adapte le tri selection
        # en se limitant a k iterations.
        j_min = i
        for j in range(i+1, len(t)):
            # j_min = indice du minimum parmi t[i],...,t[j-1]
            n = len(p)-1
            if distance_euclidienne(p,t[j],n) < distance_euclidienne(p,t[j_min],n):
                j_min = j
        if j_min != i: # on echange les cases i et j_min
            t[i], t[j_min] = t[j_min], t[i]
    return [t[i] for i in range(k)]

    def prediction(training, point, k):
    """
    Entrées:
    - training: tableau de tableaux contenant les d caractéristiques des fleurs
    - point: tableau des d-1 ou d caractéristiques du point inconnu    
    - k: entier, correspondant au nombre de voisins du calcul. 

    Renvoie:
    la classe la plus courante parmi les k plus proches voisins de point dans training.

    NB: la valeur de la d-ieme coordonnee de point (l'etiquette eventuelle) 
    n'est pas utilisee par la fonction.
    """

    # Vérification que le point inconnu a la même structure que ceux de training 
    d = len(training[0])
    if len(point) != d: # si necessaire pour harmoniser les dimensions
        point = point + [None] # completer point avec une classe fictive

    # Calcul du tableau des plus proches voisins
    voisins = plus_proches_voisins2(training, point, k)
    print(f'Liste de voisins: {voisins}')
    # Calcul de l'occurence de chaque classe parmi les voisins
    nb_occurrences = {}
    for t in voisins:
        classe = t[d-1]
        nb_occurrences[classe] = nb_occurrences.get(classe, 0) + 1

    # Recherche de la classe la plus représentée
    print(f'nb_occurrences: {nb_occurrences}')
    nb_max = 0
    cle_max = None
    for cle, valeur in nb_occurrences.items():
        if valeur > nb_max:
            cle_max = cle
            nb_max = valeur
    return cle_max

    Syntaxe

    L'utilisation de .get (que l'on appelle une méthode ... cours de terminale) avec le dictionnaire nb_occurences permet de récupérer la valeur associée à la clé classe. Si jamais cette clé n'existe pas alors cela renverra une valeur par défaut (ici 0). Cela évite un KeyError en cas d'absence de clé.
