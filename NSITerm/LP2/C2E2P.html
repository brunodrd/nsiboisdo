from typing import List


def mat2tab_v2(m: List[list]) -> list:
    """
    renvoie la matrice m sous la forme d'un tableau
    m: tableau de tableaux d'entiers
    """
    nb_element = len(m)
    a_plat = []
    for i in range(nb_element):
        for valeur in m[i]:
            a_plat.append(valeur)
    return a_plat
