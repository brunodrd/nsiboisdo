    from random import choice


def itineraire_aleatoire(m, n):
    """
    /!\ INCOHERENCES
    Ici,
    m: entier naturel correspondant à la largeur
    n: entier naturel correspondant à la longueur
    Une grille (n X m) aura des coordonnées allant respectivement
    0 <= j <= n-1 et 0 <= i <= m-1
    """
    itineraire = ''
    i, j = 0, 0
    while i != m-1 and j != n-1:
        d = choice(['D', 'B']) # il y a plusieurs lignes
        itineraire += d
        if d == 'D':
            j += 1
        else:
            i += 1
    if i == m-1:
        itineraire = itineraire + 'D'*(n-j)
    if j == n-1:
        itineraire = itineraire + 'B'*(m-i)
    return itineraire

    Question 7⚓︎

    Incohérence (encore)

    À partir d'ici m correspond à la longueur et n à la largeur !!!!

    N(m, n) est le nombre de chemins distincts de longueur m et de largeur n. Les itinéraires de longueur 1 sont obtenus par l'appel (incohérent) itineraire_aleatoire(n, 1). Dans un tel cas, on ne passe pas dans la boucle while et on vérifie systématiquement le 2ème test (if j == n-1). De ce fait il n'y a qu'un seul chemin possible. Voir les tests ci-dessous.

    Question 8⚓︎

    Pour arriver en E, on ne peut venir que du dessus ou alors de la gauche. De ce fait, pour une grille \(m\times n\), le nombre de chemins distincts est égal à la somme des chemins distincts de la grille \((m-1)\times n\) (gauche) et ceux de la grille \(m\times (n-1)\) (dessus).
    Autrement dit: N(m, n) = N(m-1, n) + N(m, n-1).

    def chercher_parent(racine, dossier):
    """
    Renvoie le parent de dossier;
    racine, dossier: instances de la classe Dossier
    """
    for f in racine.fils:
        if f.contient(dossier):
            if f.nom == dossier.nom:
                return racine
            else:
                return chercher_parent(f, dossier)
    return None

    class Dossier:
    def __init__(self, nom, liste, parent=None):
        self.nom = nom
        self.fils = liste # liste d'objets de la classe Dossier
        if parent is not None:
            self.parent = parent

    def est_vide(self):
        return self.fils == []

    def parcours(self):
        print(self.nom)
        if self.fils:
            for f in self.fils:
                f.parcours()

    def mkdir(self, nomrep):
        self.fils.append(Dossier(nomrep, [], self))

    def contient(self, nom_dossier):    

        def visites(dossier, liste):
            """
            Construit récursivement la listes des noms de dossiers
            """
            liste.append(dossier.nom)
            for f in dossier.fils:
                visites(f, liste)

        dossiers = []
        visites(self, dossiers)
        return nom_dossier in dossiers

    def __str__(self):
        return self.nom
