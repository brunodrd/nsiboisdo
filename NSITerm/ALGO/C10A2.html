    # ----------------------------------------------- #
    # Reconstruction d'un alignement de score maximal #
    # On redescend du haut vers le bas                #
    # On parcourt les choix successifs  √† rebours     #
    # Jusqu'√† ce que i ou j soit nul (cas de base)    #
    # ----------------------------------------------- #

    # alignement[0] est un alignement optimal pour seq1
    # alignement[1] est un alignement optimal pour seq2
    alignement = ['', '']    
    while i != 0 and j != 0:
        s1 = score[i - 1][j - 1]
        s2 = score[i][j - 1]
        s3 =  score[i - 1][j]
        if seq1[i - 1] == seq2[j - 1]:
            d = 1
        else:
            d = -1
        maxi = max(d + s1, -1 + s2, -1 + s3)
        if maxi == d + s1:
            alignement[0] = seq1[i - 1] + alignement[0]
            alignement[1] =  seq2[j - 1] + alignement[1]
            i = i - 1
            j = j - 1
        elif maxi == -1 + s2:            
            alignement[0] = '_' + alignement[0]
            alignement[1] =  seq2[j - 1] + alignement[1]
            j = j - 1
        else:
            alignement[0] = seq1[i - 1] + alignement[0]
            alignement[1] =  '_'  + alignement[1]
            i = i - 1
    # si i diff√©rent de 0 on aligne seq1[:i] avec i * '_' 
    if i != 0:
        alignement[0] = seq1[:i]  + alignement[0]
        alignement[1] = i * '_' + alignement[1]
    # si j diff√©rent de 0 on aligne j * '_'  avec seq2[:j]
    if j != 0:
        alignement[0] = j * '_'   + alignement[0]
        alignement[1] = seq2[:j] + alignement[1]
    return (score[n][m], alignement)
